import os
import struct
import time
import matplotlib.pyplot as plt
import numpy as np
from matplotlib.patches import Circle
from pynq import Overlay, allocate

MAX_ITER = 800
MAX_OBS = 20
MAP_SIZE = 500.0

def _float_bits(value):
    """Safely converts a Python float to a 32-bit unsigned integer bit pattern"""
    return struct.unpack("<I", struct.pack("<f", float(value)))[0]

class _RegisterAccess:
    """Dynamically finds exact AXI offsets from the IP Dictionary to prevent reading/writing empty memory."""
    def __init__(self, overlay, ip_name):
        self.ip = getattr(overlay, ip_name)
        self.registers = overlay.ip_dict.get(ip_name, {}).get("registers", {})
        
        if not self.registers:
            raise RuntimeError(
                f"\n[CRITICAL ERROR] PYNQ found '{ip_name}' but its register map is empty.\n"
                "-> Ensure 'imbkatwo.hwh' is in the exact same folder as 'imbkatwo.bit'."
            )

    def _description(self, requested):
        if requested in self.registers:
            return self.registers[requested]

        matches = [
            value for name, value in self.registers.items()
            if name.startswith(requested)
        ]
        if len(matches) >= 1:
            return matches[0]
        raise KeyError(f"Register '{requested}' not found in HWH dictionary.")

    def write(self, name, value, is_pointer=False):
        desc = self._description(name)
        offset = int(desc["address_offset"])
        raw = int(value)
        
        self.ip.write(offset, raw & 0xFFFFFFFF)
        
        if is_pointer:
            self.ip.write(offset + 4, 0)

    def read(self, name):
        desc = self._description(name)
        offset = int(desc["address_offset"])
        return int(self.ip.read(offset))

class BKAPlannerUI:
    def __init__(
        self,
        bitfile="imbkathree.bit", 
        ip_name="bka_hw_engine_0",
        start=(30.0, 30.0),
        goal=(420.0, 420.0),
    ):
        print(f"Loading Overlay: {bitfile}...")
        self.overlay = Overlay(bitfile)
        
        if not hasattr(self.overlay, ip_name):
            raise AttributeError(f"IP {ip_name} not found. Available IP: {', '.join(self.overlay.ip_dict)}")
                    
        self.regs = _RegisterAccess(self.overlay, ip_name)

        self.start = np.asarray(start, dtype=np.float32)
        self.goal = np.asarray(goal, dtype=np.float32)
        self.obstacles = []
        self.path = None
        self.drag_center = None
        self.preview = None

        print("Allocating DDR Memory...")
        self.obs_buffer = allocate(shape=(MAX_OBS * 3,), dtype=np.float32)
        self.path_x_buffer = allocate(shape=(MAX_ITER,), dtype=np.float32)
        self.path_y_buffer = allocate(shape=(MAX_ITER,), dtype=np.float32)

        print("Initializing UI...")
        self.figure, self.axes = plt.subplots(figsize=(7, 7))
        self.figure.canvas.mpl_connect("button_press_event", self._on_press)
        self.figure.canvas.mpl_connect("motion_notify_event", self._on_motion)
        self.figure.canvas.mpl_connect("button_release_event", self._on_release)
        self.figure.canvas.mpl_connect("key_press_event", self._on_key)
        
        self._draw("Drag to add obstacles | Enter to run | Backspace to undo")
        plt.show()

    def close(self):
        self.obs_buffer.close()
        self.path_x_buffer.close()
        self.path_y_buffer.close()
        plt.close(self.figure)

    def _inside_map(self, event):
        return (
            event.inaxes is self.axes
            and event.xdata is not None
            and event.ydata is not None
            and 0.0 <= event.xdata <= MAP_SIZE
            and 0.0 <= event.ydata <= MAP_SIZE
        )

    def _on_press(self, event):
        if event.button != 1 or not self._inside_map(event):
            return
        if len(self.obstacles) >= MAX_OBS:
            self._draw(f"Maximum of {MAX_OBS} obstacles reached")
            return
        self.drag_center = np.array([event.xdata, event.ydata], dtype=np.float32)
        self.preview = 3.0
        self._draw("Drag outward to set obstacle radius")

    def _on_motion(self, event):
        if self.drag_center is None or not self._inside_map(event):
            return
        point = np.array([event.xdata, event.ydata], dtype=np.float32)
        self.preview = float(np.linalg.norm(point - self.drag_center))
        self.preview = float(np.clip(self.preview, 3.0, 200.0))
        self._draw(f"Radius: {self.preview:.1f} px")

    def _on_release(self, event):
        if self.drag_center is None:
            return
        radius = 3.0 if self.preview is None else self.preview
        self.obstacles.append((
            float(self.drag_center[0]),
            float(self.drag_center[1]),
            float(radius),
        ))
        self.drag_center = None
        self.preview = None
        self.path = None
        self._draw(f"Obstacle added; {len(self.obstacles)} total")

    def _on_key(self, event):
        if event.key in ("backspace", "delete"):
            if self.obstacles:
                self.obstacles.pop()
                self.path = None
                self._draw(f"Removed latest obstacle; {len(self.obstacles)} remain")
        elif event.key in ("enter", "return"):
            self.run()
        elif event.key in ("c", "C"):
            self.obstacles.clear()
            self.path = None
            self._draw("All obstacles cleared")

    def _draw(self, status, math_stats=None):
        self.axes.clear()
        self.axes.set_xlim(0, MAP_SIZE)
        self.axes.set_ylim(0, MAP_SIZE)
        self.axes.set_aspect("equal", adjustable="box")
        self.axes.grid(True, alpha=0.25)
        self.axes.set_xlabel("X (pixels)")
        self.axes.set_ylabel("Y (pixels)")
        self.axes.set_title(status, fontweight="bold")

        # Draw Obstacles
        for x, y, radius in self.obstacles:
            self.axes.add_patch(Circle(
                (x, y), radius,
                facecolor="tab:red", edgecolor="darkred", alpha=0.35))

        # Draw Drag Preview
        if self.drag_center is not None:
            self.axes.add_patch(Circle(
                tuple(self.drag_center), float(self.preview),
                facecolor="tab:orange", edgecolor="darkorange", alpha=0.3))

        # Draw Start and Goal
        self.axes.scatter(
            [self.start[0]], [self.start[1]],
            c="tab:green", s=90, marker="o", label="Start", zorder=5)
        self.axes.scatter(
            [self.goal[0]], [self.goal[1]],
            c="tab:blue", s=110, marker="*", label="Goal", zorder=5)

        # Draw FPGA Path
        if self.path is not None and len(self.path):
            self.axes.plot(
                self.path[:, 0], self.path[:, 1],
                color="black", linewidth=2.0, label="FPGA path", zorder=4)

        # Apply the Math Data directly to the screen in a box
        if math_stats is not None:
            self.axes.text(
                0.02, 0.98, math_stats, 
                transform=self.axes.transAxes,
                fontsize=10, verticalalignment='top',
                bbox=dict(boxstyle='round,pad=0.5', facecolor='white', edgecolor='black', alpha=0.85),
                zorder=10
            )

        self.axes.legend(loc="lower right")
        self.figure.canvas.draw_idle()

    def _load_buffers(self):
        self.obs_buffer[:] = 0.0
        for index, obstacle in enumerate(self.obstacles):
            base = index * 3
            self.obs_buffer[base:base + 3] = obstacle
        self.path_x_buffer[:] = 0.0
        self.path_y_buffer[:] = 0.0
        self.obs_buffer.sync_to_device()
        self.path_x_buffer.sync_to_device()
        self.path_y_buffer.sync_to_device()

    def run(self, timeout_seconds=0.5):
        self._draw("Executing hardware acceleration...")
        
        self._load_buffers()

        self.regs.write("start_x", _float_bits(self.start[0]))
        self.regs.write("start_y", _float_bits(self.start[1]))
        self.regs.write("goal_x", _float_bits(self.goal[0]))
        self.regs.write("goal_y", _float_bits(self.goal[1]))
        self.regs.write("num_obstacles", len(self.obstacles))

        self.regs.write("obs_data", self.obs_buffer.physical_address, is_pointer=True)
        self.regs.write("path_out_x", self.path_x_buffer.physical_address, is_pointer=True)
        self.regs.write("path_out_y", self.path_y_buffer.physical_address, is_pointer=True)

        begin = time.perf_counter()
        self.regs.ip.write(0x00, 0x01) 
        
        deadline = begin + timeout_seconds
        while (self.regs.ip.read(0x00) & 0x02) == 0: 
            if time.perf_counter() > deadline:
                raise TimeoutError("Hardware engine did not finish in {:.0f} ms".format(timeout_seconds * 1000.0))
        elapsed_ms = (time.perf_counter() - begin) * 1000.0

        iterations = int(self.regs.read("iterations_used"))
        reached = int(self.regs.read("goal_reached_flag")) != 0
        
        if iterations < 1 or iterations > MAX_ITER:
            raise RuntimeError(f"Hardware returned invalid iteration count {iterations}. Register Map read failed.")

        self.path_x_buffer.sync_from_device()
        self.path_y_buffer.sync_from_device()
        
        self.path = np.column_stack((
            np.asarray(self.path_x_buffer[:iterations]).copy(),
            np.asarray(self.path_y_buffer[:iterations]).copy(),
        ))

        # --- Calculate Mathematics ---
        final_error = float(np.linalg.norm(self.path[-1] - self.goal.astype(np.float32)))
        
        path_length = 0.0
        if len(self.path) > 1:
            diffs = np.diff(self.path, axis=0)
            path_length = np.sum(np.linalg.norm(diffs, axis=1))

        # --- Format Stats Box ---
        math_str = (
            f"Mathematical Results:\n"
            f"-------------------\n"
            f"Total Length : {path_length:.1f} px\n"
            f"Distance Off : {final_error:.1f} px\n"
            f"Iterations   : {iterations}\n"
            f"HW Time      : {elapsed_ms:.3f} ms"
        )
        
        status_msg = f"{'[SUCCESS] Path Planned Successfully' if reached else '[FAILED] No Feasible Path'}"
 
        # Pass both the title and the math box to be drawn
        self._draw(status_msg, math_stats=math_str)
